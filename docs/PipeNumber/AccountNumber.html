def validate_variants
  @variants.select! do | variant |
    variant_error = AccountNumber.validate(variant)
    variant_error.nil?
  end
  if error_exists? and @variants.any?
    @value = @variants.shift
    @error = AccountNumber.validate(@value)
  end      
end
